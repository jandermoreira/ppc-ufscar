% ppc@CaixaAtividade: desenha a caixa da atividade
% #1: id da atividade
\NewDocumentCommand{\ppc@CaixaAtividade}{ m }{%
    \ATAttributeIfEqual[#1]{caráter}{Obrigatório}{%
        \colorlet{cor caixa}{cor caixa obrigatória}%
        \colorlet{cor texto}{cor texto obrigatória}%
        \colorlet{cor horas}{cor horas obrigatória}%
        \colorlet{cor gradiente}{cor gradiente obrigatória}%
    }{%
        \ATAttributeIfEqual[#1]{caráter}{Optativo}{%
            \colorlet{cor caixa}{cor caixa optativa}%
            \colorlet{cor texto}{cor texto optativa}%
            \colorlet{cor horas}{cor horas optativa}%
            \colorlet{cor gradiente}{cor gradiente optativa}%
        }{%
            \ATAttributeGetTo[#1]{\ppg@local@Carater}{caráter}
            \PackageError{ppc-ufscar}{Caráter inválido para #1 (\ppg@local@Carater); válidos: obrigatório ou optativo.}
        }%
    }%
    \begin{tikzpicture}
        \node[caixa atividade] (caixa atividade) {};
        \node[nome atividade, xshift = -0.2cm] at ($(caixa atividade.south)!0.65!(caixa atividade.north)$) (nome atividade) { \ATAttributeIfExist[#1]{nome abreviado}{\PPCAtividade{#1}{nome abreviado}}{\PPCAtividade{#1}{nome}}};
        \coordinate (horas) at ($(caixa atividade.south west) + (0, 0.05cm)$);
        \foreach \prefixo/\horas in {
            T:/teóricas, P:/práticas, Ext:/extensionistas, Est:/estágio%
        }{
            \node[horas, \horas] at (horas) (ultima hora) {\PPCAtividade{#1}{horas \horas}};
            \node[prefixo] at (ultima hora.north west) {\prefixo};
            \coordinate (horas) at (ultima hora.south east);
        }
        \ppc@ObtenhaCargaAtividade{\ppc@local@Carga}{#1}
        \PPCIf{MostraTotal}{
            \node[horas, anchor = south west, text = cor texto, scale = 0.95, align = right, inner xsep = 0.1cm] at (ultima hora.north west) (ultima hora) {\ppc@local@Carga h};
        }{}
        % \node[prefixo] at (ultima hora.north west) {Tot:};
    \end{tikzpicture}%
}

% ppc@CaixaArea: desenha a caixa para uma área específica, incluindo todos os períodos
% #1: (opcional) nome da matriz
% #2: id da área
\newcounter{ppc@local@Linha}
\newcounter{ppc@local@Coluna}
\newcounter{ppc@local@Maximo}
\NewDocumentCommand{\ppc@CaixaArea}{ O{matriz-principal} m }{%
    \begin{tikzpicture}
        \pgfdeclarelayer{atividades-curriculares}
        \pgfdeclarelayer{áreas}
        \pgfsetlayers{áreas, atividades-curriculares, main}

        % Atividades
        \begin{pgfonlayer}{atividades-curriculares}
            \defcounter{ppc@local@Maximo}{0}
            \defcounter{ppc@local@Linha}{0}
            \providebool{ppc@ApresenteOptativa}
            \PPCParaCadaPeriodo[#1]{\periodo}{
                \defcounter{ppc@local@Coluna}{0}
                \ATListForEach{#1-#2-\periodo}{\atividade}{
                    \ifstrequal{#2}{periodo}{
                        \ATAttributeIfEqual[\atividade]{caráter}{Obrigatório}{
                            \booltrue{ppc@ApresenteOptativa}
                        }{
                            \boolfalse{ppc@ApresenteOptativa}
                        }
                    }{
                        \booltrue{ppc@ApresenteOptativa}
                    }
                    \ifbool{ppc@ApresenteOptativa}{
                        \node[nó limpo] at (\theppc@local@Coluna * \ppc@LarguraCaixaAtividade + \theppc@local@Coluna * \ppc@SeparacaoCaixasAtividade, \theppc@local@Linha * -\ppc@AlturaCaixaAtividade - \theppc@local@Linha * \ppc@SeparacaoCaixasAtividade
                        ) {\ppc@CaixaAtividade{\atividade}};
                        \global\stepcounter{ppc@local@Coluna}
                    }{}
                }
                \ifstrequal{#2}{periodo}{
                    \ATListForEach{#1-carga-optativas-\periodo}{\cargahoraria}{
                        \PPCDefinaAtividade{ppc@optativa-fantasma}{
                            horas teóricas = \cargahoraria,
                        }
                        \node[nó limpo] at (\theppc@local@Coluna * \ppc@LarguraCaixaAtividade + \theppc@local@Coluna * \ppc@SeparacaoCaixasAtividade, \theppc@local@Linha * -\ppc@AlturaCaixaAtividade - \theppc@local@Linha * \ppc@SeparacaoCaixasAtividade
                        ) {\ppc@CaixaAtividade{ppc@optativa-fantasma}};
                        \global\stepcounter{ppc@local@Coluna}
                    }
                }{}
                \ifnumcomp{\theppc@local@Coluna}{>}{\theppc@local@Maximo}{
                    \global\defcounter{ppc@local@Maximo}{\theppc@local@Coluna}
                }{}
                \global\stepcounter{ppc@local@Linha}
            }
            \global\addtocounter{ppc@local@Linha}{-1} % volta 1 na contagem
        \end{pgfonlayer}

        \begin{pgfonlayer}{áreas}
            % caixa com as atividades
            \coordinate (caixa area sup esq) at (-\ppc@LarguraCaixaAtividade/2 - \ppc@SeparacaoCaixasAtividade, \ppc@AlturaCaixaAtividade/2 + \ppc@SeparacaoCaixasAtividade);
            \coordinate (caixa area inf dir) at (\theppc@local@Maximo * \ppc@LarguraCaixaAtividade - \ppc@LarguraCaixaAtividade/2 + \theppc@local@Maximo * \ppc@SeparacaoCaixasAtividade, -\theppc@local@Linha * \ppc@AlturaCaixaAtividade - \theppc@local@Linha * \ppc@SeparacaoCaixasAtividade - \ppc@AlturaCaixaAtividade/2 - \ppc@SeparacaoCaixasAtividade);
            \node[nó limpo, nome área, fit = (caixa area sup esq)(caixa area inf dir)] (caixa area) {};

            % caixa com o nome da área
            \PPCIf{MostraNomeArea}{
                \coordinate (caixa area inf esq) at (-\ppc@LarguraCaixaAtividade/2 - \ppc@SeparacaoCaixasAtividade, \ppc@AlturaCaixaAtividade/2 + 2\ppc@SeparacaoCaixasAtividade);
                \coordinate (caixa area sup dir) at (\theppc@local@Maximo * \ppc@LarguraCaixaAtividade - \ppc@LarguraCaixaAtividade/2 + \theppc@local@Maximo * \ppc@SeparacaoCaixasAtividade, \ppc@AlturaNomeArea + \ppc@AlturaCaixaAtividade/2 + 2\ppc@SeparacaoCaixasAtividade);
                \node[nó limpo, nome área, fit=(caixa area inf esq)(caixa area sup dir)] (título) {};
                \node[nome área, inner sep = 0cm, text width = \theppc@local@Maximo * \ppc@LarguraCaixaAtividade + \theppc@local@Maximo * \ppc@SeparacaoCaixasAtividade + \ppc@SeparacaoCaixasAtividade - 0.5em] at (título) {\PPCArea[#1]{#2}{nome}};
            }{}
        \end{pgfonlayer}
    \end{tikzpicture}%
}


% ppc@CaixaPeriodos: desenha a caixa a coluna de períodos
% #1: (opcional) nome da matriz
\NewDocumentCommand{\ppc@CaixaPeriodos}{ O{matriz-principal} }{%
    \begin{tikzpicture}
        \pgfdeclarelayer{períodos}
        \pgfdeclarelayer{coluna}
        \pgfsetlayers{coluna, períodos, main}

        % Períodos
        \begin{pgfonlayer}{períodos}
            \defcounter{ppc@local@Linha}{0}
            \PPCParaCadaPeriodo[#1]{\periodo}{
                \node[nó limpo] at (0, \theppc@local@Linha * -\ppc@AlturaCaixaAtividade - \theppc@local@Linha * \ppc@SeparacaoCaixasAtividade) {
                    \ppc@CaixaPeriodo[#1]{\periodo}
                };
                \global\stepcounter{ppc@local@Linha}
            }
            \global\addtocounter{ppc@local@Linha}{-1} % volta 1 na contagem
        \end{pgfonlayer}

        \begin{pgfonlayer}{coluna}
            % caixa com os períodos
            \coordinate (caixa area sup esq) at (-\ppc@LarguraCaixaAtividade/4 - \ppc@SeparacaoCaixasAtividade, \ppc@AlturaCaixaAtividade/2 + \ppc@SeparacaoCaixasAtividade);
            \coordinate (caixa area inf dir) at (\ppc@LarguraCaixaAtividade/4 + \ppc@SeparacaoCaixasAtividade, -\theppc@local@Linha * \ppc@AlturaCaixaAtividade - \theppc@local@Linha * \ppc@SeparacaoCaixasAtividade - \ppc@AlturaCaixaAtividade/2 - \ppc@SeparacaoCaixasAtividade);
            \node[nó limpo, nome área, fit = (caixa area sup esq)(caixa area inf dir)] (caixa area) {};

            \PPCIf{MostraNomeArea}{
                \coordinate (caixa area inf esq) at (-\ppc@LarguraCaixaAtividade/4 - \ppc@SeparacaoCaixasAtividade, \ppc@AlturaCaixaAtividade/2 + 2\ppc@SeparacaoCaixasAtividade);
                \coordinate (caixa area sup dir) at (\ppc@LarguraCaixaAtividade/4 + \ppc@SeparacaoCaixasAtividade, \ppc@AlturaNomeArea + \ppc@AlturaCaixaAtividade/2 + 2\ppc@SeparacaoCaixasAtividade);
                \node[nó limpo, nome área, fit=(caixa area inf esq)(caixa area sup dir)] (título) {};
                \node[nome área, inner sep = 0cm, text width = \ppc@LarguraCaixaAtividade/2 + 2\ppc@SeparacaoCaixasAtividade - 1em] at (título) {Total: \PPCCargaHoraria[#1]{total}[\,h]};
            }{}
        \end{pgfonlayer}
    \end{tikzpicture}%
}


% ppc@CaixaMatriz: desenha a caixa da matriz curricular completa
% #1: (opcional) nome da matriz
% #2: id da área
\NewDocumentCommand{\ppc@CaixaMatriz}{ O{matriz-principal} }{%
    \resizebox{\linewidth}{!}{
        \begin{tikzpicture}%[transform shape, scale = 0.5]
            \PPCIf{MostraColunaPeriodos}{
                \node[caixa de área] (períodos) {\ppc@CaixaPeriodos[#1]};
                \coordinate (referência) at ($(períodos.north east) + (\ppc@SeparacaoCaixasAtividade, 0)$);
            }{
                \coordinate (referência) at (0, 0);
            }
            \PPCIf{VisaoDoAluno}{
                \node[caixa de área] at (referência) (área) {\ppc@CaixaArea[#1]{periodo}};
            }{
                \ATListForEach{#1-áreas}{\area}{
                    \node[caixa de área] at (referência) (área) {\ppc@CaixaArea[#1]{\area}};
                    \coordinate (referência) at ($(área.north east) + (\ppc@SeparacaoCaixasAtividade, 0)$);
                }
            }
        \end{tikzpicture}
    }
}
